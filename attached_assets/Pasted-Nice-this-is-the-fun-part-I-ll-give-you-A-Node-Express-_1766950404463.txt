3. API route for /api/contact-form/submit
// server/contact-form-route.ts (or inside your main server file)

import type { Request, Response } from "express";
import { google } from "googleapis";
import nodemailer from "nodemailer";

/**
 * Helper: create Google Sheets client with service account
 */
async function getSheetsClient() {
  const clientEmail = process.env.GOOGLE_SERVICE_ACCOUNT_EMAIL;
  const privateKeyRaw = process.env.GOOGLE_SERVICE_ACCOUNT_PRIVATE_KEY;

  if (!clientEmail || !privateKeyRaw) {
    console.warn("Google Sheets env vars missing, skipping Sheets integration.");
    return null;
  }

  const privateKey = privateKeyRaw.replace(/\\n/g, "\n");

  const auth = new google.auth.JWT({
    email: clientEmail,
    key: privateKey,
    scopes: ["https://www.googleapis.com/auth/spreadsheets"],
  });

  await auth.authorize();

  return google.sheets({ version: "v4", auth });
}

/**
 * Helper: create Nodemailer transporter for emails
 */
function getMailTransporter() {
  const { SMTP_HOST, SMTP_PORT, SMTP_USER, SMTP_PASS } = process.env;

  if (!SMTP_HOST || !SMTP_PORT || !SMTP_USER || !SMTP_PASS) {
    console.warn("SMTP env vars missing, skipping email sending.");
    return null;
  }

  return nodemailer.createTransport({
    host: SMTP_HOST,
    port: Number(SMTP_PORT) || 587,
    secure: Number(SMTP_PORT) === 465, // true for 465, false otherwise
    auth: {
      user: SMTP_USER,
      pass: SMTP_PASS,
    },
  });
}

/**
 * Very simple template replacer for {{name}}, {{email}}, etc.
 */
function applyTemplate(template: string, formData: Record<string, any>) {
  if (!template) return template;
  return template.replace(/{{\s*([\w.-]+)\s*}}/g, (_, key) => {
    const value = formData[key];
    return value != null ? String(value) : "";
  });
}

/**
 * The actual route handler
 */
export async function contactFormSubmitHandler(req: Request, res: Response) {
  try {
    const { formData = {}, formConfig = {}, meta } = req.body || {};

    if (!formData || typeof formData !== "object") {
      return res.status(400).json({ success: false, message: "Missing formData" });
    }

    // ---- 1) GOOGLE SHEETS: append row ----
    let sheetOk = false;

    try {
      if (formConfig.googleSheetsEnabled) {
        const sheets = await getSheetsClient();
        if (sheets) {
          const spreadsheetId =
            formConfig.googleSheetsSpreadsheetId ||
            process.env.GOOGLE_SHEETS_SPREADSHEET_ID;

          const range =
            formConfig.googleSheetsRange ||
            process.env.GOOGLE_SHEETS_DEFAULT_RANGE ||
            "Sheet1!A1";

          if (spreadsheetId) {
            // Decide column order:
            const fieldConfigs = formConfig.fieldConfigs || [];
            let keys: string[] = [];

            if (Array.isArray(fieldConfigs) && fieldConfigs.length) {
              keys = fieldConfigs
                .filter((f: any) => f.enabled)
                .map((f: any) => f.key)
                .filter(Boolean);
            } else {
              keys = Object.keys(formData);
            }

            const rowValues = keys.map((k) =>
              formData[k] != null ? String(formData[k]) : ""
            );

            // Optional: push meta as JSON string at the end
            if (meta) {
              rowValues.push(JSON.stringify(meta));
            }

            await sheets.spreadsheets.values.append({
              spreadsheetId,
              range,
              valueInputOption: "RAW",
              insertDataOption: "INSERT_ROWS",
              requestBody: {
                values: [rowValues],
              },
            });

            sheetOk = true;
          }
        }
      }
    } catch (err) {
      console.error("Google Sheets append error:", err);
      // continue; don't fail the whole request
    }

    // ---- 2) AUTO-REPLY EMAIL TO USER ----
    let autoReplyOk = false;
    let adminMailOk = false;

    try {
      const transporter = getMailTransporter();
      if (transporter) {
        const mailFrom =
          formConfig.fromEmail ||
          process.env.MAIL_FROM ||
          process.env.SMTP_USER;

        // Auto-reply to the person who filled the form
        if (formConfig.autoReplyEnabled) {
          const toFieldKey = formConfig.autoReplyToFieldKey || "email";
          const toAddress = formData[toFieldKey];

          if (toAddress) {
            const subject =
              formConfig.autoReplySubject || "Thanks for contacting us!";
            const rawMessage =
              formConfig.autoReplyMessage ||
              "Hi {{name}},\n\nThank you for your message. We will get back to you soon.\n\nBest,\nYour Team";

            const text = applyTemplate(rawMessage, formData);

            await transporter.sendMail({
              from: mailFrom,
              to: toAddress,
              subject,
              text,
            });

            autoReplyOk = true;
          }
        }

        // Optional: send an email to admin with form submission
        if (formConfig.notifyAdminEmail) {
          const subject = formConfig.notifyAdminSubject || "New contact form submission";
          const lines = Object.entries(formData).map(
            ([key, value]) => `${key}: ${value}`
          );
          const metaLine = meta ? `\nMeta: ${JSON.stringify(meta, null, 2)}` : "";

          await transporter.sendMail({
            from: mailFrom,
            to: formConfig.notifyAdminEmail,
            subject,
            text: lines.join("\n") + metaLine,
          });

          adminMailOk = true;
        }
      }
    } catch (err) {
      console.error("Email (auto-reply/admin) error:", err);
      // also non-fatal
    }

    return res.json({
      success: true,
      googleSheets: sheetOk,
      autoReply: autoReplyOk,
      adminEmail: adminMailOk,
    });
  } catch (error) {
    console.error("contact-form/submit fatal error:", error);
    return res.status(500).json({
      success: false,
      message: "Internal server error",
    });
  }
}

Tiny fix for your from_name is not defined error

In your React editor UI, somewhere you probably wrote:

Message (supports {{name}}, {{from_name}})


JSX thinks {from_name} is a JavaScript variable. Change it to a plain string like:

<label className="text-xs text-gray-400 block mb-1">
  Message (supports {"{{name}}"}, {"{{from_name}}"})
</label>


Now it’s safe.

1️⃣ Create server/routes/contact-form.ts

If your backend uses plain JS, I’ll show a JS version after this.
For TypeScript, create server/routes/contact-form.ts and paste:

// server/routes/contact-form.ts
import express, { Request, Response } from "express";
import { google } from "googleapis";
import nodemailer from "nodemailer";

const router = express.Router();

/**
 * Helper: apply {{placeholders}} using formData/meta
 */
function applyTemplate(
  template: string,
  values: Record<string, any> = {}
): string {
  if (!template) return "";
  return template.replace(/{{\s*([^}]+)\s*}}/g, (_, key) => {
    const v = values[key];
    if (v === undefined || v === null) return "";
    return String(v);
  });
}

/**
 * Helper: init Google Sheets client using service account
 */
async function getSheetsClient() {
  const clientEmail = process.env.GOOGLE_SERVICE_ACCOUNT_EMAIL;
  const privateKeyRaw = process.env.GOOGLE_SERVICE_ACCOUNT_PRIVATE_KEY;

  if (!clientEmail || !privateKeyRaw) {
    throw new Error("Google service account env vars are missing");
  }

  const privateKey = privateKeyRaw.replace(/\\n/g, "\n");

  const auth = new google.auth.JWT(
    clientEmail,
    undefined,
    privateKey,
    ["https://www.googleapis.com/auth/spreadsheets"]
  );

  await auth.authorize();
  return google.sheets({ version: "v4", auth });
}

/**
 * Helper: init mail transporter
 */
function getMailTransporter() {
  const host = process.env.SMTP_HOST;
  const port = Number(process.env.SMTP_PORT || 587);
  const user = process.env.SMTP_USER;
  const pass = process.env.SMTP_PASS;

  if (!host || !user || !pass) {
    throw new Error("SMTP env vars are missing");
  }

  return nodemailer.createTransport({
    host,
    port,
    secure: false, // change to true if you use 465
    auth: { user, pass },
  });
}

/**
 * POST /api/contact-form/submit
 */
router.post(
  "/submit",
  async (req: Request, res: Response): Promise<void> => {
    try {
      const { formData, formConfig, meta } = req.body || {};

      if (!formData || typeof formData !== "object") {
        res.status(400).json({ success: false, message: "Invalid formData" });
        return;
      }

      // -----------------------------
      // 1. Read automation settings from visual editor config
      // -----------------------------
      const automation = (formConfig?.automation || {}) as any;

      // Google Sheets
      const googleSheetsEnabled: boolean =
        automation.googleSheetsEnabled ?? false;
      const googleSheetsId: string =
        automation.googleSheetsId || process.env.GOOGLE_SHEETS_ID || "";
      const googleSheetsTab: string = automation.googleSheetsTab || "Sheet1";

      // Auto-reply
      const autoReplyEnabled: boolean = automation.autoReplyEnabled ?? false;
      const autoReplyEmailField: string =
        automation.autoReplyEmailField || "email"; // which field holds visitor email
      const defaultFromName =
        automation.fromName ||
        process.env.MAIL_FROM_NAME ||
        "Website Contact";
      const defaultFromEmail =
        automation.fromEmail ||
        process.env.MAIL_FROM_EMAIL ||
        process.env.SMTP_USER ||
        "";

      const autoReplySubjectTemplate: string =
        automation.autoReplySubject || "Thanks, {{name}} — we got your message";
      const autoReplyBodyTemplate: string =
        automation.autoReplyBody ||
        "Hi {{name}},\n\nThanks for reaching out. We received your message and will reply soon.\n\nBest regards,\n" +
          defaultFromName;

      // Admin notification
      const notifyOwner: boolean = automation.notifyOwner ?? true;
      const ownerEmail: string =
        automation.ownerEmail || process.env.CONTACT_OWNER_EMAIL || "";

      // -----------------------------
      // 2. Save to Google Sheets
      // -----------------------------
      if (googleSheetsEnabled && googleSheetsId) {
        try {
          const sheets = await getSheetsClient();

          // Build a flat row: timestamp + all keys
          const timestamp = new Date().toISOString();
          const keys = Object.keys(formData);
          const values = keys.map((k) =>
            typeof formData[k] === "object"
              ? JSON.stringify(formData[k])
              : formData[k]
          );

          await sheets.spreadsheets.values.append({
            spreadsheetId: googleSheetsId,
            range: `${googleSheetsTab}!A1`,
            valueInputOption: "RAW",
            insertDataOption: "INSERT_ROWS",
            requestBody: {
              values: [[timestamp, ...values]],
            },
          });
        } catch (sheetErr) {
          console.error("Google Sheets append error:", sheetErr);
          // Don't fail the whole request because of Sheets
        }
      }

      // -----------------------------
      // 3. Send emails (auto-reply + admin)
      // -----------------------------
      try {
        const transporter = getMailTransporter();

        const templateValues = {
          ...formData,
          ...(meta || {}),
        };

        // Auto-reply to visitor
        const replyToAddress = formData[autoReplyEmailField];

        if (autoReplyEnabled && replyToAddress) {
          await transporter.sendMail({
            from: `"${defaultFromName}" <${defaultFromEmail}>`,
            to: replyToAddress,
            subject: applyTemplate(autoReplySubjectTemplate, templateValues),
            text: applyTemplate(autoReplyBodyTemplate, templateValues),
          });
        }

        // Notification to site owner
        if (notifyOwner && ownerEmail) {
          const prettyForm = JSON.stringify(formData, null, 2);
          const prettyMeta = meta ? JSON.stringify(meta, null, 2) : "(none)";

          await transporter.sendMail({
            from: `"${defaultFromName}" <${defaultFromEmail}>`,
            to: ownerEmail,
            subject:
              automation.ownerSubject ||
              `New contact form submission from ${
                formData.name || formData.email || "your site"
              }`,
            text:
              `You received a new contact form submission.\n\n` +
              `Form data:\n${prettyForm}\n\n` +
              `Meta:\n${prettyMeta}\n`,
          });
        }
      } catch (mailErr) {
        console.error("Mail send error:", mailErr);
        // Also best-effort; don't block success
      }

      // -----------------------------
      // 4. Respond to frontend
      // -----------------------------
      res.json({ success: true });
    } catch (err) {
      console.error("Contact form submit error:", err);
      res
        .status(500)
        .json({ success: false, message: "Server error processing form" });
    }
  }
);

export default router;
