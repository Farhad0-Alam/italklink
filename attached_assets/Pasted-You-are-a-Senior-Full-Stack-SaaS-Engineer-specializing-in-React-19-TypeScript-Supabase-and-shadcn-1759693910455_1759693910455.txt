🧩 Database Schema (Supabase)
CREATE TABLE plans (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name text NOT NULL,
  plan_type text CHECK (plan_type IN ('Free','Paid')),
  frequency text CHECK (frequency IN ('Monthly','Yearly')),
  currency text DEFAULT 'USD',
  price numeric DEFAULT 0,
  trial_days integer DEFAULT 0,
  card_limit integer DEFAULT 1,
  base_users integer DEFAULT 1,
  price_per_user numeric DEFAULT 0,
  setup_fee numeric DEFAULT 0,
  features jsonb DEFAULT '{}',
  templates jsonb DEFAULT '[]',
  active boolean DEFAULT true,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE coupons (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  code text UNIQUE NOT NULL,
  display_name text,
  description text,
  type text CHECK (type IN ('Percentage','Fixed')),
  percentage numeric DEFAULT 0,
  max_discount numeric DEFAULT 0,
  start_date date,
  expiry_date date,
  total_usage_limit integer DEFAULT 0,
  uses_per_user integer DEFAULT 1,
  min_order_amount numeric DEFAULT 0,
  applicable_plans uuid[] DEFAULT '{}',
  active boolean DEFAULT true,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE user_subscriptions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid REFERENCES auth.users(id),
  plan_id uuid REFERENCES plans(id),
  coupon_id uuid REFERENCES coupons(id),
  user_count integer DEFAULT 1,
  price_paid numeric DEFAULT 0,
  start_date timestamptz DEFAULT now(),
  end_date timestamptz,
  active boolean DEFAULT true,
  features jsonb DEFAULT '{}'
);
