import { apiGet } from "@/lib/api";
import { useQuery } from "@tanstack/react-query";
import { Button } from "@/components/ui/button";

export function UserOrdersPage() {
  const { data, isLoading } = useQuery({
    queryKey: ["user-orders"],
    queryFn: () => apiGet("/api/user/orders"),
  });

  return (
    <div className="p-5 space-y-5">
      <h1 className="text-xl font-bold">My Orders</h1>

      {isLoading ? (
        <p>Loading orders...</p>

        <p>No orders yet.</p>
      ) : (
        <table className="w-full text-sm border-collapse">
          <thead className="bg-muted/40">
            <tr>
              <th className="px-3 py-2 text-left">Product</th>
              <th className="px-3 py-2 text-left">Amount</th>
              <th className="px-3 py-2 text-left">Date</th>
              <th className="px-3 py-2 text-right">Download</th>
            </tr>
          </thead>

          <tbody>
            {data.map((order) => (
              <tr key={order.id} className="border-t">
                <td className="px-3 py-2">{order.product_title}</td>
                <td className="px-3 py-2">
                  {order.amount} {order.currency}
                </td>
                <td className="px-3 py-2">
                  {new Date(order.created_at).toLocaleDateString()}
                </td>
                <td className="px-3 py-2 text-right">
                  <Button
                    size="sm"
                    variant="outline"
                    onClick={() =>
                      window.open(`/api/download/${order.download_token}`)
                    }
                  >
                    Download
                  </Button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      )}
    </div>
  );
}

import express from "express";
import { query } from "../lib/db.js";
import { authRequired } from "../lib/auth.js";

const router = express.Router();

router.get("/", authRequired, async (req, res) => {
  try {
    const sql = `
      SELECT 
        o.id,
        o.amount,
        o.currency,
        o.created_at,
        o.download_token,
        p.title AS product_title
      FROM orders o
      JOIN products p ON p.id = o.product_id
      WHERE o.buyer_id = $1
      ORDER BY o.created_at DESC
    `;

    const result = await query(sql, [req.user.id]);
    res.json(result.rows);

  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

🟦 6. SECURE FILE DOWNLOAD ROUTE

(From Replit filesystem)

server/routes/download.js

import express from "express";
import fs from "fs";
import path from "path";
import { query } from "../lib/db.js";

const router = express.Router();

router.get("/:token", async (req, res) => {
  const { token } = req.params;

  const sql = `
    SELECT d.file_path, d.download_count, d.max_downloads, d.expires_at
    FROM downloads d
    WHERE d.download_token = $1
    LIMIT 1
  `;

  const result = await query(sql, [token]);

  if (result.rows.length === 0) {
    return res.status(404).json({ error: "Invalid token" });
  }

  const row = result.rows[0];
  const filePath = row.file_path;

  // Expiry check
  if (new Date(row.expires_at) < new Date()) {
    return res.status(403).json({ error: "Download expired" });
  }

  // Download limit check
  if (row.download_count >= row.max_downloads) {
    return res.status(403).json({ error: "Download limit reached" });
  }

  // Update count
  await query(
    "UPDATE downloads SET download_count = download_count + 1 WHERE download_token = $1",
    [token]
  );

  // Full file path
  const fullPath = path.join(process.cwd(), filePath);

  if (!fs.existsSync(fullPath)) {
    return res.status(404).json({ error: "File missing" });
  }

  res.download(fullPath);
});

export default router;
