🟥 File: src/pages/user/orders.tsx
import { apiGet } from "@/lib/api";
import { useQuery } from "@tanstack/react-query";
import { Button } from "@/components/ui/button";

export function UserOrdersPage() {
  const { data, isLoading } = useQuery({
    queryKey: ["user-orders"],
    queryFn: () => apiGet("/api/user/orders"),
  });

  return (
    <div className="p-5 space-y-5">
      <h1 className="text-xl font-bold">My Orders</h1>

      {isLoading ? (
        <p>Loading orders...</p>
      ) : data?.length === 0 ? (
        <p>No orders yet.</p>
      ) : (
        <table className="w-full text-sm border-collapse">
          <thead className="bg-muted/40">
            <tr>
              <th className="px-3 py-2 text-left">Product</th>
              <th className="px-3 py-2 text-left">Amount</th>
              <th className="px-3 py-2 text-left">Date</th>
              <th className="px-3 py-2 text-right">Download</th>
            </tr>
          </thead>

          <tbody>
            {data.map((order) => (
              <tr key={order.id} className="border-t">
                <td className="px-3 py-2">{order.product_title}</td>
                <td className="px-3 py-2">
                  {order.amount} {order.currency}
                </td>
                <td className="px-3 py-2">
                  {new Date(order.created_at).toLocaleDateString()}
                </td>
                <td className="px-3 py-2 text-right">
                  <Button
                    size="sm"
                    variant="outline"
                    onClick={() =>
                      window.open(`/api/download/${order.download_token}`)
                    }
                  >
                    Download
                  </Button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      )}
    </div>
  );
}

import { apiGet } from "@/lib/api";
import { useQuery } from "@tanstack/react-query";
import { Button } from "@/components/ui/button";

export function UserDownloadsPage() {
  const { data, isLoading } = useQuery({
    queryKey: ["user-downloads"],
    queryFn: () => apiGet("/api/user/downloads"),
  });

  return (
    <div className="p-5 space-y-5">
      <h1 className="text-xl font-bold">My Downloads</h1>

      {isLoading ? (
        <p>Loading downloads...</p>
      ) : data?.length === 0 ? (
        <p>You haven't downloaded any items yet.</p>
      ) : (
        <table className="w-full text-sm">
          <thead className="bg-muted/40">
            <tr>
              <th className="px-3 py-2 text-left">File</th>
              <th className="px-3 py-2 text-left">Expires</th>
              <th className="px-3 py-2 text-left">Left</th>
              <th className="px-3 py-2 text-right">Download</th>
            </tr>
          </thead>
          <tbody>
            {data.map((d) => (
              <tr key={d.id} className="border-t">
                <td className="px-3 py-2">{d.product_title}</td>
                <td className="px-3 py-2">
                  {new Date(d.expires_at).toLocaleDateString()}
                </td>
                <td className="px-3 py-2">
                  {d.max_downloads - d.download_count}
                </td>
                <td className="px-3 py-2 text-right">
                  <Button
                    size="sm"
                    variant="outline"
                    onClick={() =>
                      window.open(`/api/download/${d.download_token}`)
                    }
                    disabled={d.max_downloads - d.download_count <= 0}
                  >
                    Download
                  </Button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      )}
    </div>
  );
}
