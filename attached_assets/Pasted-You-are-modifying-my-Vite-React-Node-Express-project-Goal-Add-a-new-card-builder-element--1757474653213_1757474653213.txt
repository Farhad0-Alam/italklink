export async function compileMind(file: File) {
  const fd = new FormData();
  fd.append("image", file);
  const r = await fetch("/api/ar/compile", { method: "POST", body: fd });
  const d = await r.json();
  if (!r.ok || !d.ok) throw new Error(d.error || "Compile failed");
  return d as { ok: true; mindFileUrl: string; textureUrl?: string };
}

4) Server: Cloudinary helper (server/modules/ar/cloudinary.ts)
ts

import fetch from "node-fetch";

const CLOUDINARY_URL = process.env.CLOUDINARY_URL!; // cloudinary://key:secret@cloud
if (!CLOUDINARY_URL) {
  // We won't crash; we'll throw only when used
}

function parseCloudinaryUrl() {
  const m = CLOUDINARY_URL.match(/^cloudinary:\/\/([^:]+):([^@]+)@([^/]+)$/);
  if (!m) throw new Error("Invalid CLOUDINARY_URL");
  return { key: m[1], secret: m[2], cloud: m[3] };
}

export async function uploadImageToCloudinary(filename: string, buffer: Buffer) {
  if (!CLOUDINARY_URL) throw new Error("CLOUDINARY_URL not set");
  const { key, secret, cloud } = parseCloudinaryUrl();
  const url = `https://api.cloudinary.com/v1_1/${cloud}/image/upload`;
  const ts = Math.floor(Date.now()/1000);
  const form = new FormData();
  form.append("file", new Blob([buffer]), filename);
  form.append("timestamp", String(ts));
  form.append("api_key", key);
  form.append("upload_preset", ""); // optional preset; leave blank if you sign manually

  // unsigned is tricky; sign the request:
  const toSign = new URLSearchParams({ timestamp: String(ts) }).toString();
  const crypto = await import("crypto");
  const signature = crypto.createHash("sha1").update(toSign + secret).digest("hex");
  form.append("signature", signature);

  const r = await fetch(url, { method: "POST", body: form as any });
  const d = await r.json();
  if (!r.ok) throw new Error(d?.error?.message || "cloudinary_image_failed");
  return d.secure_url as string;
}

export async function uploadRawToCloudinary(filename: string, buffer: Buffer) {
  if (!CLOUDINARY_URL) throw new Error("CLOUDINARY_URL not set");
  const { key, secret, cloud } = parseCloudinaryUrl();
  const url = `https://api.cloudinary.com/v1_1/${cloud}/raw/upload`;
  const ts = Math.floor(Date.now()/1000);
  const form = new FormData();
  form.append("file", new Blob([buffer]), filename);
  form.append("timestamp", String(ts));
  form.append("api_key", key);

  const toSign = new URLSearchParams({ timestamp: String(ts) }).toString();
  const crypto = await import("crypto");
  const signature = crypto.createHash("sha1").update(toSign + secret).digest("hex");
  form.append("signature", signature);

  const r = await fetch(url, { method: "POST", body: form as any });
  const d = await r.json();
  if (!r.ok) throw new Error(d?.error?.message || "cloudinary_raw_failed");
  return d.secure_url as string;
}

6) Server: compile handler (server/modules/ar/compile.ts)
ts
Copy code
import fetch from "node-fetch";
import { uploadImageToCloudinary, uploadRawToCloudinary } from "./cloudinary";

const PROXY = process.env.AR_COMPILER_PROXY_URL;

export async function handleCompile(imageBuffer: Buffer, originalName: string) {
  // 1) Upload the original image (for planeTextureUrl + record)
  const textureUrl = await uploadImageToCloudinary(originalName || "card-front.jpg", imageBuffer);

  if (!PROXY) {
    return {
      ok: false,
      status: 501,
      error: "Auto-compile not configured. Set AR_COMPILER_PROXY_URL or paste a .mind URL manually.",
      textureUrl
    };
  }

  // 2) Send to proxy compiler
  const fd = new FormData();
  fd.append("image", new Blob([imageBuffer]), originalName || "card-front.jpg");

  const r = await fetch(PROXY, { method: "POST", body: fd as any });
  const d = await r.json().catch(() => ({}));

  if (!r.ok || !d?.ok) {
    return { ok: false, status: 502, error: d?.error || "compiler_failed", textureUrl };
  }

  if (d.mindUrl) {
    // Proxy already hosted a .mind file
    return { ok: true, mindFileUrl: d.mindUrl as string, textureUrl };
  }

  if (d.mindBufferBase64) {
    // 3) Upload compiled .mind to Cloudinary as raw
    const buf = Buffer.from(d.mindBufferBase64, "base64");
    const mindFileUrl = await uploadRawToCloudinary("targets.mind", buf);
    return { ok: true, mindFileUrl, textureUrl };
  }

  return { ok: false, status: 500, error: "invalid_compiler_response", textureUrl };
}
7) Server: routes (server/modules/ar/routes.ts)

Copy code
import { Router } from "express";
import { upload } from "./upload";
import { handleCompile } from "./compile";

const r = Router();

r.post("/api/ar/compile", upload.single("image"), async (req, res) => {
  try {
    if (!req.file?.buffer) return res.status(400).json({ ok:false, error:"image required" });
    const result = await handleCompile(req.file.buffer, req.file.originalname);
    if (!result.ok) return res.status(result.status || 500).json(result);
    res.json({ ok: true, mindFileUrl: result.mindFileUrl, textureUrl: result.textureUrl });
  } catch (e:any) {
    res.status(500).json({ ok:false, error: e.message || "error" });
  }
});

export default r;
