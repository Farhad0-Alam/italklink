Frontend:
// src/pages/admin/users.tsx
import { useQuery, useMutation } from "@tanstack/react-query";
import { apiGet, apiPost } from "@/lib/api";
import { Button } from "@/components/ui/button";

export function AdminUsersPage() {
  const { data, isLoading } = useQuery({
    queryKey: ["admin-users"],
    queryFn: () => apiGet("/api/admin/users"),
  });

  const disableMutation = useMutation({
    mutationFn: (id: string) =>
      apiPost("/api/admin/users/disable", { id }),
  });

  const enableMutation = useMutation({
    mutationFn: (id: string) =>
      apiPost("/api/admin/users/enable", { id }),
  });

  return (
    <div className="p-6 space-y-4">
      <h1 className="text-xl font-bold">All Users</h1>

      {isLoading ? (
        <p>Loading…</p>
      ) : (
        <table className="w-full border-collapse text-sm">
          <thead className="bg-muted/40 text-xs text-muted-foreground">
            <tr>
              <th className="px-4 py-3 text-left">Email</th>
              <th className="px-4 py-3 text-left">Status</th>
              <th className="px-4 py-3 text-right">Actions</th>
            </tr>
          </thead>
          <tbody>
            {data?.map((u) => (
              <tr key={u.id} className="border-t hover:bg-muted/20">
                <td className="px-4 py-3">{u.email}</td>
                <td className="px-4 py-3">
                  {u.disabled ? "Disabled" : "Active"}
                </td>
                <td className="px-4 py-3 text-right space-x-2">
                  {!u.disabled ? (
                    <Button
                      variant="destructive"
                      size="sm"
                      onClick={() => disableMutation.mutate(u.id)}
                    >
                      Disable
                    </Button>
                  ) : (
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => enableMutation.mutate(u.id)}
                    >
                      Enable
                    </Button>
                  )}
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      )}
    </div>
  );
}

🟣 1) Admin Middleware (admin only)

/server/lib/auth.js

export function isAdmin(req, res, next) {
  try {
    if (!req.user) {
      return res.status(401).json({ error: "Unauthorized" });
    }

    if (req.user.role !== "admin") {
      return res.status(403).json({ error: "Admins only" });
    }

    next();
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
}

🟨 5) ADMIN ANALYTICS ROUTES

/server/routes/admin.analytics.js

import express from "express";
import { query } from "../lib/db.js";
import { isAdmin } from "../lib/auth.js";

const router = express.Router();

/**
 * PLATFORM ANALYTICS
 */
router.get("/", isAdmin, async (req, res) => {
  try {
    // Total Revenue
    const revenue = await query(`SELECT COALESCE(SUM(amount), 0) AS total_revenue FROM orders`);

    // Total Sales
    const sales = await query(`SELECT COUNT(*) AS total_sales FROM orders`);

    // Users count
    const users = await query(`SELECT COUNT(*) AS total_users FROM users`);

    // Active products
    const products = await query(`SELECT COUNT(*) AS active_products FROM products WHERE status='active'`);

    // Top Products
    const topProducts = await query(`
      SELECT p.id, p.title, COUNT(o.id) AS total_sales
      FROM products p
      JOIN orders o ON o.product_id = p.id
      GROUP BY p.id
      ORDER BY total_sales DESC
      LIMIT 10
    `);

    res.json({
      total_revenue: revenue.rows[0].total_revenue,
      total_sales: sales.rows[0].total_sales,
      total_users: users.rows[0].total_users,
      active_products: products.rows[0].active_products,
      top_products: topProducts.rows,
    });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});

export default router;
