import OneSignal from "react-onesignal";

/** Call once in app bootstrap (e.g., main.tsx or App.tsx) */
export async function initOneSignal() {
  await OneSignal.init({
    appId: import.meta.env.VITE_ONESIGNAL_APP_ID,
    allowLocalhostAsSecureOrigin: true,
    notifyButton: { enable: true },
  });

  try {
    const user = (window as any).__AUTH?.user;
    if (user) {
      OneSignal.sendTag("role", "user");
      OneSignal.sendTag("owner_user_id", String(user.id));
      OneSignal.sendTag("plan", user.plan ?? "free");
    }
    const locale = navigator.language?.split("-")[0] || "en";
    OneSignal.sendTag("locale", locale);
  } catch {
    // ignore tagging failures
  }
}

/** Tag a visitor on a public card page */
export function tagCardVisitor(card: { id: string | number; slug?: string }) {
  OneSignal.sendTag("audience", "visitor");
  OneSignal.sendTag("card_id", String(card.id));
  if (card.slug) OneSignal.sendTag("card_slug", card.slug);
}

/** Optional helper to prompt subscription */
export function promptPush() {
  return OneSignal.Slidedown.promptPush();
}

  const [url, setUrl] = useState("");

  async function send() {
    setLoading(true);
    setErr(null);
    try {
      const r = await fetch("/api/notify/card", {
        method: "POST",

/** naive in-memory rate limit */
const buckets = new Map<string, { count: number; resetAt: number }>();

export function rateLimit(key: string, limit: number) {
  const now = Date.now();
  const rec = buckets.get(key);
  if (!rec || rec.resetAt < now) {
    buckets.set(key, { count: 1, resetAt: now + 24 * 60 * 60 * 1000 });
    return { ok: true };
  }
  if (rec.count >= limit) return { ok: false, retryAt: rec.resetAt };
  rec.count += 1;
  return { ok: true };
}

import fetch from "node-fetch";

const ONE_URL = "https://api.onesignal.com/notifications";

type SendArgs = {
  headings: Record<string, string>;
  contents: Record<string, string>;
  url?: string;
  filters: any[];
  send_after?: string; // ISO
};

export async function sendByTags(args: SendArgs) {
  const body = {
    app_id: process.env.ONESIGNAL_APP_ID,
    ...args,
  };

  const r = await fetch(ONE_URL, {
    method: "POST",
    headers: {
      Authorization: `Basic ${process.env.ONESIGNAL_API_KEY}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify(body),
  });

  const data = await r.json();
  if (!r.ok) {
    const msg = (data && (data.errors || data.error)) || "onesignal_failed";
    throw new Error(typeof msg === "string" ? msg : JSON.stringify(msg));
  }
  return data;
}

import { Request, Response } from "express";
import { sendByTags } from "./service";
import { rateLimit } from "./mw";

export async function notifyCardSubscribers(req: Request, res: Response) {
  try {
    const user = (req as any).user || { id: "u" };
    const { cardId, title, message, url } = req.body || {};
    if (!cardId || !title || !message) {
      return res.status(400).json({ ok: false, error: "cardId, title, message required" });
    }

    const key = `card:${user.id}:${cardId}`;
    const rl = rateLimit(key, 3);
    if (!rl.ok) return res.status(429).json({ ok: false, error: "rate_limited" });

    const data = await sendByTags({
      headings: { en: String(title) },
      contents: { en: String(message) },
      url,
      filters: [{ field: "tag", key: "card_id", relation: "=", value: String(cardId) }],
    });

    res.json({ ok: true, data });
  } catch (e: any) {
    res.status(500).json({ ok: false, error: e.message || "error" });
  }
}

export async function notifyAdminBroadcast(req: Request, res: Response) {
  try {
    const user = (req as any).user || { id: "admin" };
    const { title, message, url, segment = "all", locales, countries } = req.body || {};
    if (!title || !message) return res.status(400).json({ ok: false, error: "title, message required" });

    const key = `admin:${user.id}`;
    const rl = rateLimit(key, 2);
    if (!rl.ok) return res.status(429).json({ ok: false, error: "rate_limited" });

    const filters: any[] = [{ field: "tag", key: "role", relation: "=", value: "user" }];

    if (segment === "paid") {
      filters.push({ operator: "OR" });
      filters.push({ field: "tag", key: "plan", relation: "=", value: "pro" });
      filters.push({ operator: "OR" });
      filters.push({ field: "tag", key: "plan", relation: "=", value: "premium" });
    } else if (segment === "free") {
      filters.push({ operator: "AND" });
      filters.push({ field: "tag", key: "plan", relation: "=", value: "free" });
    }

    const pushIn = (arr?: string[], key?: string) => {
      if (!arr || !arr.length || !key) return;
      // Build (key IN arr) as OR chain
      filters.push({ operator: "AND" });
      arr.forEach((v, i) => {
        if (i > 0) filters.push({ operator: "OR" });
        filters.push({ field: "tag", key, relation: "=", value: v });
      });
    };

    pushIn(locales, "locale");
    pushIn(countries, "country");

    const data = await sendByTags({
      headings: { en: String(title) },
      contents: { en: String(message) },
      url,
      filters,
    });

    res.json({ ok: true, data });
  } catch (e: any) {
    res.status(500).json({ ok: false, error: e.message || "error" });
  }
}
